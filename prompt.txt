    function getMoonPhase() {
      const date = new Date();
      const year = date.getFullYear();
      const month = date.getMonth() + 1;
      const day = date.getDate();
      let c, e, jd, b;
      if (month < 3) { year--; month += 12; }
      c = 365.25 * year;
      e = 30.6 * month;
      jd = c + e + day - 694039.09;
      jd /= 29.5305882;
      b = parseInt(jd);
      jd -= b;
      b = Math.round(jd * 8);
      if (b >= 8) b = 0;
      const phases = ["New Moon", "Waxing Crescent", "First Quarter", "Waxing Gibbous", "Full Moon", "Waning Gibbous", "Last Quarter", "Waning Crescent"];
      return phases[b];
    }

    function Starfield() {
      const canvasRef = useRef(null);
      useEffect(() => {
        const canvas = canvasRef.current;
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        const geometry = new THREE.BufferGeometry();
        const count = 6000;
        const positions = new Float32Array(count * 3);
        for (let i = 0; i < count * 3; i++) {
          positions[i] = (Math.random() - 0.5) * 40;
        }
        geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
        const material = new THREE.PointsMaterial({ color: 0xffffff, size: 0.03, transparent: true, opacity: 0.8, blending: THREE.AdditiveBlending, sizeAttenuation: true });
        const stars = new THREE.Points(geometry, material);
        scene.add(stars);
        camera.position.z = 5;
        let animId;
        const animate = () => {
          animId = requestAnimationFrame(animate);
          stars.rotation.x += 0.0003;
          stars.rotation.y += 0.0006;
          renderer.render(scene, camera);
        };
        animate();
        const onResize = () => {
          camera.aspect = window.innerWidth / window.innerHeight;
          camera.updateProjectionMatrix();
          renderer.setSize(window.innerWidth, window.innerHeight);
        };
        window.addEventListener('resize', onResize);
        return () => {
          cancelAnimationFrame(animId);
          window.removeEventListener('resize', onResize);
          renderer.dispose();
          geometry.dispose();
          material.dispose();
        };
      }, []);
      return <canvas ref={canvasRef} className="star-canvas" />;
    }
    
    // --- MAIN APP ---
    function App() {
      const [scene, setScene] = useState('arrival');
      const [profile, setProfile] = useState(() => {
        try { return JSON.parse(localStorage.getItem('soulmirror_profile') || '{}'); } catch(e) { return {}; }
      });
      const [journal, setJournal] = useState(() => {
        try { return JSON.parse(localStorage.getItem('soulmirror_journal') || '[]'); } catch(e) { return []; }
      });
      const [currentGuide, setCurrentGuide] = useState(null);
      const [sessionId, setSessionId] = useState(null);
      const [crisisActive, setCrisisActive] = useState(false);
      
      const saveProfile = (p) => {
        const final = { ...p, sunSign: p.birthDate ? getSunSign(p.birthDate) : null };
        setProfile(final);
        localStorage.setItem('soulmirror_profile', JSON.stringify(final));
      };
      
      const addJournal = (entry) => {
        const item = { id: Date.now(), date: new Date().toISOString(), ...entry };
        const updated = [item, ...journal];
        setJournal(updated);
        localStorage.setItem('soulmirror_journal', JSON.stringify(updated));
      };
      
      const emotionalWeather = useMemo(() => {
        if (!journal.length) return 'calm';
        const last = journal[0].emotionalWeather || 'calm';
        return last;
      }, [journal]);
      
      const handleCrisis = () => {
        setCrisisActive(true);
        setScene('crisis');
      };
      
      const enterChamber = (guideId) => {
        setCurrentGuide(guideId);
        setSessionId('sess_' + Date.now());
        setScene('chamber');
      };
      
      const enterMirror = () => {
        setSessionId('sess_' + Date.now());
        setScene('mirror');
      };
      
      if (scene === 'arrival') return <Arrival onEnter={() => setScene('onboarding')} />;
      if (scene === 'onboarding') return <Onboarding onComplete={(p) => { saveProfile(p); setScene('hall'); }} />;
      if (scene === 'crisis') return <SafeHarbor onResolve={() => setScene('hall')} profile={profile} />;
      if (scene === 'hall') return <GreatHall profile={profile} emotionalWeather={emotionalWeather} onEnterChamber={enterChamber} onEnterMirror={enterMirror} onJournal={() => setScene('journal')} />;
      if (scene === 'chamber') return <Chamber guideId={currentGuide} profile={profile} onBack={() => setScene('hall')} onCrisis={handleCrisis} onSaveToJournal={addJournal} />;
      if (scene === 'mirror') return <MirrorCouncil profile={profile} onBack={() => setScene('hall')} onCrisis={handleCrisis} onSaveToJournal={addJournal} />;
      if (scene === 'journal') return <SoulJournal entries={journal} onClose={() => setScene('hall')} profile={profile} />;
      return <div>Unknown scene</div>;
    }

    function Chamber({ guideId, profile, onBack, onCrisis, onSaveToJournal }) {
      const guide = GUIDES[guideId];

      const [input, setInput] = useState('');
      const [loading, setLoading] = useState(false);
      const [typingText, setTypingText] = useState('');
      const bottomRef = useRef(null);
      
      const chamberClass = `chamber-${guideId}`;
      
      useEffect(() => {
        const greeting = `Welcome, seeker. I am the ${guide.name}. ${guideId === 'stoic' ? 'Sit beside the pool. Let us look clearly at what is.' : guideId === 'nurturing' ? 'Come closer to the fire. You are safe here.' : guideId === 'coach' ? 'Step into the forge. We have work to do.' : guideId === 'astrologer' ? 'Look up. The stars have been waiting to speak with you.' : guideId === 'alchemist' ? 'Descend with me. The gold is hidden in the darkness.' : 'The roses are blooming. Sit with me in the twilight.'}`;
        typeMessage(greeting);
        setMessages([{role:'guide', content:greeting}]);
      }, []);
      
      useEffect(() => { bottomRef.current?.scrollIntoView({behavior:'smooth'}); }, [messages, typingText]);
      
      const typeMessage = (text) => {
        setTypingText('');
        let i = 0;
        const interval = setInterval(() => {
          i++;
          setTypingText(text.slice(0, i));
          if (i >= text.length) clearInterval(interval);
        }, 20);
        return () => clearInterval(interval);
      };
      
      const handleSend = () => {
        if (!input.trim() || loading) return;
        const userMsg = input.trim();
        setMessages(prev => [...prev, {role:'seeker', content:userMsg}]);
        setInput('');
        
        if (checkCrisis(userMsg)) {
          onCrisis();
          return;
        }
        
        setLoading(true);
        setTimeout(() => {
          let resp;
          if (guideId === 'stoic') resp = generateStoicResponse(userMsg, profile);
          else if (guideId === 'nurturing') resp = generateNurturingResponse(userMsg, profile);
          else if (guideId === 'coach') resp = generateCoachResponse(userMsg, profile);
          else if (guideId === 'astrologer') resp = generateAstrologerResponse(userMsg, profile);
          else if (guideId === 'alchemist') resp = generateAlchemistResponse(userMsg, profile);
          else resp = generatePoetResponse(userMsg, profile);
          
          const content = resp.content;
          setMessages(prev => [...prev, {role:'guide', content, metadata: resp}]);
          setLoading(false);
          onSaveToJournal({ guideId, guideName: guide.name, query: userMsg, response: content, emotionalWeather: 'calm', entryType: 'reflection' });
          typeMessage(content);
        }, 1200);
      };
      
      return (
        <div className={`relative w-full min-h-screen ${chamberClass} overflow-hidden flex flex-col`}>
          <div className="absolute inset-0 bg-black/40" />

                </div>

```typescript
import { authMiddleware } from "@clerk/nextjs";
import { NextResponse } from "next/server";

export default authMiddleware({
  publicRoutes: ["/", "/sign-in(.*)", "/sign-up(.*)", "/api/webhooks/clerk"],
  
  async afterAuth(auth, req) {
    // If user is signed in and trying to access public routes, redirect to /hall
    if (auth.userId && (req.nextUrl.pathname === "/" || 
        req.nextUrl.pathname.startsWith("/sign-in") || 
        req.nextUrl.pathname.startsWith("/sign-up"))) {
      const hallUrl = new URL("/hall", req.url);
      return NextResponse.redirect(hallUrl);
    }
    
    // If user is not signed in and trying to access protected routes, redirect to sign-in
    if (!auth.userId && !auth.isPublicRoute) {
      const signInUrl = new URL("/sign-in", req.url);
      signInUrl.searchParams.set("redirect_url", req.url);
      return NextResponse.redirect(signInUrl);
    }
  }
});
```

Clerk Webhook for User Sync:

```typescript
// app/api/webhooks/clerk/route.ts
import { Webhook } from "svix";
import { headers } from "next/headers";
import { WebhookEvent } from "@clerk/nextjs/server";
import { prisma } from "@/lib/prisma";

export async function POST(req: Request) {
  const WEBHOOK_SECRET = process.env.CLERK_WEBHOOK_SECRET;
  if (!WEBHOOK_SECRET) throw new Error("Missing CLERK_WEBHOOK_SECRET");

  const headerPayload = headers();
  const svix_id = headerPayload.get("svix-id");
  const svix_timestamp = headerPayload.get("svix-timestamp");
  const svix_signature = headerPayload.get("svix-signature");

  if (!svix_id || !svix_timestamp || !svix_signature) {
    return new Response("Missing svix headers", { status: 400 });
  }

  const payload = await req.json();
  const body = JSON.stringify(payload);
  const wh = new Webhook(WEBHOOK_SECRET);
  
  let evt: WebhookEvent;
  try {
    evt = wh.verify(body, {
      "svix-id": svix_id,
      "svix-timestamp": svix_timestamp,
      "svix-signature": svix_signature,
    }) as WebhookEvent;
  } catch (err) {
    return new Response("Invalid signature", { status: 400 });
  }

  const eventType = evt.type;

  if (eventType === "user.created" || eventType === "user.updated") {
    const { id, email_addresses, first_name, last_name } = evt.data;
    const primaryEmail = email_addresses?.[0]?.email_address;

    await prisma.user.upsert({
      where: { clerkId: id },
      update: {
        email: primaryEmail || "",
        name: `${first_name || ""} ${last_name || ""}`.trim(),
      },
      create: {
        clerkId: id,
        email: primaryEmail || "",
        name: `${first_name || ""} ${last_name || ""}`.trim(),
      },
    });
  }

  if (eventType === "user.deleted") {
    const { id } = evt.data;
    if (id) {
      await prisma.user.delete({ where: { clerkId: id } }).catch(() => {});
    }
  }

  return new Response("OK", { status: 200 });
}
```

Guide Response Endpoint (Single Guide):

```typescript
// app/api/guides/[guideId]/respond/route.ts
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@clerk/nextjs";
import { prisma } from "@/lib/prisma";
import { getGuideSystemPrompt } from "@/lib/guides/prompts";
import { saveMessage } from "@/lib/services/messages";
import { analyzeCrisis } from "@/lib/services/crisis-detection";

export async function POST(
  req: NextRequest,
  { params }: { params: { guideId: string } }
) {
  const { userId } = auth();
  if (!userId) return new NextResponse("Unauthorized", { status: 401 });

  const user = await prisma.user.findUnique({ where: { clerkId: userId } });
  if (!user) return new NextResponse("User not found", { status: 404 });

  const { sessionId, content } = await req.json();
  const guideId = params.guideId;

  // Crisis detection
  const crisisDetected = await analyzeCrisis(content);
  if (crisisDetected) {
    // Trigger Safe Harbor protocol
    await triggerCrisisProtocol(user.id, sessionId);
    return NextResponse.json({ crisis: true });
  }

  // Save seeker's message
  await saveMessage({
    sessionId,
    role: "seeker",
    content,
  });

  // Get conversation history
  const history = await prisma.message.findMany({
    where: { sessionId },
    orderBy: { createdAt: "desc" },
    take: 20,
  });

  // Get guide's system prompt
  const systemPrompt = getGuideSystemPrompt(guideId, user);

  // Call OpenRouter
  const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.OPENROUTER_API_KEY}`,
      "HTTP-Referer": process.env.APP_URL || "http://localhost:3000",
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      model: "openai/gpt-4o",
      messages: [
        { role: "system", content: systemPrompt },
        ...history.reverse().map(m => ({
          role: m.role === "seeker" ? "user" : "assistant",
          content: m.content,
        })),
        { role: "user", content },
      ],
      response_format: { type: "json_object" }, // Force structured output
      temperature: 0.7,
    }),
  });

  const data = await response.json();
  const guideResponse = JSON.parse(data.choices[0].message.content);

  // Save guide's response
  const savedMessage = await saveMessage({
    sessionId,
    role: `guide_${guideId}`,
    guideId,
    content: guideResponse.content,
    metadata: guideResponse.metadata,
  });

  // Update session title if first message
  const messageCount = await prisma.message.count({ where: { sessionId } });
  if (messageCount <= 2) {
    const title = await generateSessionTitle(content);
    await prisma.session.update({
      where: { id: sessionId },
      data: { title },
    });
  }

  return NextResponse.json({
    message: savedMessage,
    guideResponse,
  });
}
```

Council Synthesis Endpoint:

```typescript
// app/api/mirror/synthesize/route.ts
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@clerk/nextjs";
import { prisma } from "@/lib/prisma";
import { getGuideSystemPrompt } from "@/lib/guides/prompts";
import { SYNTHESIZER_PROMPT } from "@/lib/guides/synthesizer-prompt";

export async function POST(req: NextRequest) {
  const { userId } = auth();
  if (!userId) return new NextResponse("Unauthorized", { status: 401 });

  const user = await prisma.user.findUnique({ where: { clerkId: userId } });
  const { sessionId, content, guides } = await req.json();

  // Save seeker's message
  await prisma.message.create({
    data: { sessionId, role: "seeker", content },
  });

  // Get context
  const history = await prisma.message.findMany({
    where: { sessionId },
    orderBy: { createdAt: "desc" },
    take: 20,
  });

  // Call all guides in parallel
  const guidePromises = guides.map(async (guideId: string) => {
    const systemPrompt = getGuideSystemPrompt(guideId, user);
    
    const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${process.env.OPENROUTER_API_KEY}`,
        "HTTP-Referer": process.env.APP_URL || "http://localhost:3000",
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        model: "openai/gpt-4o",
        messages: [
          { role: "system", content: systemPrompt },
          ...history.reverse().map(m => ({
            role: m.role === "seeker" ? "user" : "assistant",
            content: m.content,
          })),
          { role: "user", content },
        ],
        response_format: { type: "json_object" },
      }),
    });

    const data = await response.json();
    const parsed = JSON.parse(data.choices[0].message.content);
    
    // Save individual guide response
    await prisma.message.create({
      data: {
        sessionId,
        role: `guide_${guideId}`,
        guideId,
        content: parsed.content,
        metadata: parsed.metadata,
      },
    });

    return { guideId, ...parsed };
  });

  const guideResponses = await Promise.all(guidePromises);

  // Synthesize
  const synthesisResponse = await fetch("https://openrouter.ai/api/v1/chat/completions", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.OPENROUTER_API_KEY}`,
      "HTTP-Referer": process.env.APP_URL || "http://localhost:3000",
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      model: "openai/gpt-4o",
      messages: [
        { role: "system", content: SYNTHESIZER_PROMPT },
        {
          role: "user",
          content: JSON.stringify({
            seekerQuery: content,
            guideResponses,
          }),
        },
      ],
      response_format: { type: "json_object" },
    }),
  });

  const synthesisData = await synthesisResponse.json();
  const synthesis = JSON.parse(synthesisData.choices[0].message.content);

  // Save synthesis
  await prisma.message.create({
    data: {
      sessionId,
      role: "synthesis",
      content: synthesis.reflection,
      metadata: synthesis,
    },
  });

  return NextResponse.json({
    guideResponses,
    synthesis,
  });
}
```

```typescript
// lib/services/crisis-detection.ts
export async function analyzeCrisis(content: string): Promise<boolean> {
  // Fast check with smaller model
  const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.OPENROUTER_API_KEY}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      model: "openai/gpt-4o-mini", // Fast, cheap model for detection
      messages: [
        {
          role: "system",
          content: `You are a crisis detection system. Analyze the user's message and determine if they are expressing:
- Active suicidal ideation or intent
- Plans for self-harm
- Disclosure of ongoing physical or sexual abuse
- Immediate danger to themselves or others

Respond with JSON: { "crisis": boolean, "severity": "low" | "medium" | "high" | "none" }`,
        },
        { role: "user", content },
      ],
      response_format: { type: "json_object" },
    }),
  });

  const data = await response.json();
  const result = JSON.parse(data.choices[0].message.content);
  
  return result.crisis && result.severity !== "none";
}

export async function triggerCrisisProtocol(userId: string, sessionId: string) {
  // 1. Log crisis event
  await prisma.crisisLog.create({
    data: {
      userId,
      sessionId,
      status: "active",
    },
  });

  // 2. Update session to lock out other guides
  await prisma.session.update({
    where: { id: sessionId },
    data: {
      activeGuides: ["safe_harbor"],
      isCouncilSession: false,
    },
  });

  // 3. Return Safe Harbor response
  return {
    crisis: true,
    safeHarborResponse: {
      content: `I hear you. And I want you to know: you are not alone right now. What you're carrying is heavy, and you deserve support that goes beyond what I can offer in this space.

If you're in immediate danger or thinking about ending your life, please reach out to a crisis service in your area right now:

These are real people, right now, who can help. You deserve that help.

I'm here with you in this moment. Take a breath with me. In... and out. You've survived every difficult moment up until now. You can survive this one too. One breath at a time.`,
      metadata: {
```

```typescript
'use client';

import { useRef, useMemo } from 'react';
import { Canvas, useFrame } from '@react-three/fiber';
import { Points, PointMaterial } from '@react-three/drei';
import * as THREE from 'three';

function StarField() {
  const ref = useRef<THREE.Points>(null!);
  
  const positions = useMemo(() => {
    const pos = new Float32Array(5000 * 3);
    for (let i = 0; i < 5000; i++) {
      pos[i * 3] = (Math.random() - 0.5) * 20;
      pos[i * 3 + 1] = (Math.random() - 0.5) * 20;
      pos[i * 3 + 2] = (Math.random() - 0.5) * 20;
    }
    return pos;
  }, []);

  useFrame((state) => {
    ref.current.rotation.x = state.clock.elapsedTime * 0.01;
    ref.current.rotation.y = state.clock.elapsedTime * 0.02;
  });

  return (
    <Points ref={ref} positions={positions} stride={3}>
      <PointMaterial
        transparent
        color="#ffffff"
        size={0.02}
        sizeAttenuation={true}
        depthWrite={false}
        blending={THREE.AdditiveBlending}
      />
    </Points>
  );
}

export function CosmicBackground() {
  return (
    <div className="fixed inset-0 -z-10">
      <Canvas camera={{ position: [0, 0, 5] }}>
        <StarField />
      </Canvas>
    </div>
  );
}
```
